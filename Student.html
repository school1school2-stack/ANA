function finish() { let score = 0; let resultsDetails = []; // لتخزين تفاصيل كل سؤال examData.forEach((item, i) => { let uVal = (item.type=="5" || item.type=="6") ? (document.querySelector(`input[name="q${i}"]:checked`)?.value || "لم يحل") : document.getElementById('ans'+i).value; const clean = (s) => s.replace(/[أإآ]/g, 'ا').replace(/ة/g, 'ه').replace(/\s+/g, '').trim(); let isCorrect = clean(uVal).includes(clean(item.ans)) && uVal.trim() !== ""; if(isCorrect) score++; resultsDetails.push({ question: item.q, userAnswer: uVal, correctAnswer: item.ans, status: isCorrect }); }); // إرسال النتيجة إلى Firebase const testID = new URLSearchParams(window.location.search).get('id') || "generic_test"; const studentName = document.getElementById('stdName').value; const studentClass = document.getElementById('stdClass').value; database.ref('results/' + testID).push({ name: studentName, class: studentClass, score: score, total: examData.length, details: resultsDetails, timestamp: firebase.database.ServerValue.TIMESTAMP }).then(() => { // عرض النتيجة للطالب كما في الكود السابق showResultToStudent(score); }); }
